local Self = nil
local PlayerTwo = nil
local PlayCylinder = nil
local PlaySphere = nil
local util = require 'xlua.util'
local tick = 0
local StartToExcute = false

local DelayToExcuteFinish = util.cs_generator(function()
		coroutine.yield(CS.UnityEngine.WaitForSeconds(5))
		Self:ProcessFinish("2")
	end)

local TargetScale = CS.UnityEngine.Vector3(0.25,2,0.25)

function AssetInit(data,target)
	print("lua AssetInit")
	local playertwo = CS.UnityEngine.Resources.Load(CS.THEDARKKNIGHT.ConstDefine.BFameWorkPathDefine.BFameResourceTestProcessPath .. "/playerTwo")
	PlayerTwo = CS.UnityEngine.GameObject.Instantiate(playertwo)
	Self = target
	Self:ReadyToExcute()
end

function DataInit(data,target)
	print("lua DataInit : " .. target.TaskName)
	PlayCylinder = PlayerTwo.transform:Find("playerTwo_2").gameObject;
	PlaySphere = PlayerTwo.transform:Find("playerTwo_3").gameObject;
	CS.THEDARKKNIGHT.EventSystem.BEventManager:Instance():AddListener(CS.THEDARKKNIGHT.EventDefine.BatEventDefine.LEFTPRESSEVENT,LeftPressCallbackPlayerTwoOne)
end

function Destory()
	print("lua Destory PlayerTwo_1")
	CS.UnityEngine.GameObject.Destroy(PlayerTwo)
	CS.THEDARKKNIGHT.EventSystem.BEventManager:Instance():RemoveListener(CS.THEDARKKNIGHT.EventDefine.BatEventDefine.LEFTPRESSEVENT,LeftPressCallbackPlayerTwoOne)
	Self = nil
	PlayerTwo = nil
	PlayCylinder = nil
	PlaySphere = nil
end

function FixedUpdate()
	--print("lua FixedUpdate")
end

function ProcessExcute()
	print("lua ProcessExcute")
end

function StopExcute()
	print("lua StopExcute")
end

function Update()
	if (StartToExcute and StartToExcute ~= nil) 
		then
		tick = tick + CS.UnityEngine.Time.deltaTime
		PlayCylinder.transform.localScale = CS.UnityEngine.Vector3.Lerp(PlayCylinder.transform.localScale, TargetScale, tick / 5)
		if (PlayCylinder.transform.localScale == TargetScale)
			then
			StartToExcute = false
			tick = 0;
			Self:ProcessFinish("1")
		end
	end
end

function LeftPressCallbackPlayerTwoOne(data)
	local packet = data
	if (packet.Info.CastGameObject == PlayCylinder)
		then
		StartToExcute = true
	elseif (packet.Info.CastGameObject == PlaySphere)
		then
		PlaySphere:AddComponent(typeof(CS.UnityEngine.Rigidbody));
		Self.mono:StartCoroutine(DelayToExcuteFinish)
	end
	return nuil
end
