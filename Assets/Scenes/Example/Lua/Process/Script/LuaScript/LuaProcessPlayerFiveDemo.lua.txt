local util = require 'xlua.util'

local Self = nil

local PlayerFive = nil

local PlayBody = nil


local DelayToExcuteFinish = util.cs_generator(function()
		coroutine.yield(CS.UnityEngine.WaitForSeconds(8))
		Self:ProcessFinish()
	end)

function AssetInit(data,target)
    print("lua AssetInit"..target.TaskName)
	local playerFiveGB = CS.UnityEngine.Resources.Load(CS.THEDARKKNIGHT.ConstDefine.BFameWorkPathDefine.BFameResourceTestProcessPath .. "/PlayerFive")
	PlayerFive = CS.UnityEngine.GameObject.Instantiate(playerFiveGB)
	Self = target	
	Self:ReadyToExcute()
end

function DataInit(data,target)
    print("lua DataInit"..target.TaskName)
	CS.THEDARKKNIGHT.EventSystem.BEventManager:Instance():AddListener(CS.THEDARKKNIGHT.EventDefine.BatEventDefine.LEFTPRESSEVENT,LeftPressCallbackPlayerFive)
end

function Destory()
    print("lua Destory")
	CS.UnityEngine.GameObject.Destroy(PlayerFive)
	CS.THEDARKKNIGHT.EventSystem.BEventManager:Instance():RemoveListener(CS.THEDARKKNIGHT.EventDefine.BatEventDefine.LEFTPRESSEVENT,LeftPressCallbackPlayerFive)
end

    --print("lua FixedUpdate")
end

    print("lua ProcessExcute")
end

    print("lua StopExcute")
end

end

function LeftPressCallbackPlayerFive(data)
	print("LeftPressCallback PlayerFive" )
    local packet = data;
    if (packet.Info.CastGameObject == PlayerFive)
	then
		PlayFiveAddRigidbody()
    end
	return nuil;
end

function PlayFiveAddRigidbody()
	print("PlayerFive.transform.childCount :"..PlayerFive.transform.childCount)
	for i = 0,PlayerFive.transform.childCount-1 do
		local child = PlayerFive.transform:GetChild(i).gameObject	
		if (child.name ~= "Option") 
		then
			print("child.name :"..child.name)
			child:AddComponent(typeof(CS.UnityEngine.Rigidbody))
			local collider = child:GetComponent(typeof(CS.UnityEngine.Collider))
			if(collider ~= nil)
			then
				collider.enabled = true
			end
		end
	end
	Self.mono:StartCoroutine(DelayToExcuteFinish)
end
