local Self = nil
local PlayerTwo = nil
local util = require 'xlua.util'
local yield_return = (require 'cs_coroutine').yield_return
local DelayToExcuteFinish = coroutine.create(function()
    print('coroutine start!')
    yield_return(CS.UnityEngine.WaitForSeconds(8))
	PlayerTwo:GetComponent("Rotate").IsNeedRotate = false;
	Self:ProcessFinish();
end)

function AssetInit(data,target)
    print("lua AssetInit")
	local Player = CS.UnityEngine.Resources.Load(CS.THEDARKKNIGHT.ConstDefine.BFameWorkPathDefine.BFameResourceTestProcessPath .. "/PlayerTwo_1")
	PlayerTwo = CS.UnityEngine.GameObject.Instantiate(Player)
	Self = target
	Self:ReadyToExcute()
end

function DataInit(data,target)
    print("lua DataInit")
	CS.THEDARKKNIGHT.EventSystem.BEventManager:Instance():AddListener(CS.THEDARKKNIGHT.EventDefine.BatEventDefine.LEFTPRESSEVENT,LeftPressCallbackPlayerTwoOne)
end

function Destory()
    print("lua Destory PlayerTwo_1")
	CS.UnityEngine.GameObject.Destroy(PlayerTwo)
	CS.THEDARKKNIGHT.EventSystem.BEventManager:Instance():RemoveListener(CS.THEDARKKNIGHT.EventDefine.BatEventDefine.LEFTPRESSEVENT,LeftPressCallbackPlayerTwoOne)
end

    --print("lua FixedUpdate")
end

    print("lua ProcessExcute")
end

function StopExcute()
    print("lua StopExcute")
end

end

function LeftPressCallbackPlayerTwoOne(data)
	print("LeftPressCallback PlayerTwo_1" )
    local packet = data;
    if (packet.Info.CastGameObject == PlayerTwo)
	then
		PlayerTwo:GetComponent("Rotate").IsNeedRotate = true;
        assert(coroutine.resume(DelayToExcuteFinish))
    end
	return nuil;
end
