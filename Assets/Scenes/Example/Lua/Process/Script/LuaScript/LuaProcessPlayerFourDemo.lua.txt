local Self = nil
local PlayerFour = nil

local TargetScale = CS.UnityEngine.Vector3(2,2,2);

local StartToExcute = false;

local float Timer = 0;


function AssetInit(data,target)
    print("lua AssetInit :"..target.TaskName)
	local playerFour = CS.UnityEngine.Resources.Load(CS.THEDARKKNIGHT.ConstDefine.BFameWorkPathDefine.BFameResourceTestProcessPath .. "/PlayerFour")
	PlayerFour = CS.UnityEngine.GameObject.Instantiate(playerFour)
	Self = target
	Self:ReadyToExcute() 
end

function DataInit(data,target)
    print("lua DataInit :"..target.TaskName)
	CS.THEDARKKNIGHT.EventSystem.BEventManager:Instance():AddListener(CS.THEDARKKNIGHT.EventDefine.BatEventDefine.LEFTPRESSEVENT,LeftPressCallbackPlayerFourDemo)
end

function Destory()
    print("lua Destory")
	CS.UnityEngine.GameObject.Destroy(PlayerFour)
	CS.THEDARKKNIGHT.EventSystem.BEventManager:Instance():RemoveListener(CS.THEDARKKNIGHT.EventDefine.BatEventDefine.LEFTPRESSEVENT,LeftPressCallbackPlayerFourDemo)
end

function FixedUpdate()
    --print("lua FixedUpdate")
end

function ProcessExcute()
    print("lua ProcessExcute")
end

function StopExcute()
    print("lua StopExcute")
end

function Update()
	if (StartToExcute)
		then
			Timer = Timer + CS.UnityEngine.Time.deltaTime;
			PlayerFour.transform.localScale = CS.UnityEngine.Vector3.Lerp(PlayerFour.transform.localScale, TargetScale, Timer / 5);
			if (PlayerFour.transform.localScale == TargetScale)
				then
					StartToExcute = false;
					Timer = 0;
					Self:ProcessFinish();
			end
	 end
end

function LeftPressCallbackPlayerFourDemo(data)
    local packet = data
    if (packet.Info.CastGameObject == PlayerFour)
	then
		StartToExcute = true
    end

end
