local Self = nil
local PlayerOne = nil
local Time = 0
local IsNeedToMove = false
local MovePosition 

function AssetInit(data,target)
    print("lua AssetInit : " .. target.TaskName)
	local Player = CS.UnityEngine.Resources.Load(CS.THEDARKKNIGHT.ConstDefine.BFameWorkPathDefine.BFameResourceTestProcessPath .. "/PlayerOne")
	PlayerOne = CS.UnityEngine.GameObject.Instantiate(Player)
	Self = target
	Self:ReadyToExcute()
end

function DataInit(data,target)
    print("lua DataInit : " .. target.TaskName)
	CS.THEDARKKNIGHT.EventSystem.BEventManager:Instance():AddListener(CS.THEDARKKNIGHT.EventDefine.BatEventDefine.LEFTPRESSEVENT,LeftPressCallbackPlayOne)
	MovePosition = PlayerOne.transform.position + PlayerOne.transform.forward * 3;
end

function Destory()
    print("lua Destory PlayerOne")
	CS.UnityEngine.GameObject.Destroy(PlayerOne)
	CS.THEDARKKNIGHT.EventSystem.BEventManager:Instance():RemoveListener(CS.THEDARKKNIGHT.EventDefine.BatEventDefine.LEFTPRESSEVENT,LeftPressCallbackPlayOne)
end

function FixedUpdate()
    --print("lua FixedUpdate")
end

function ProcessExcute()
    print("lua ProcessExcute")
end

function StopExcute()
    print("lua StopExcute")
end

function Update()
    if (IsNeedToMove)
	    then
        PlayerOne.transform.position = CS.UnityEngine.Vector3.Lerp(PlayerOne.transform.position, MovePosition, Time / 10);
        if (Time > 3)
        then
            Time = 0;
            Self:ProcessFinish();
            IsNeedToMove = false;
		else
			Time = Time + CS.UnityEngine.Time.deltaTime
		end
	end
end

function LeftPressCallbackPlayOne(data)
	print("LeftPressCallback PlayerOne" )

    if (packet.Info.CastGameObject == PlayerOne)
	then
		IsNeedToMove = true;

end
