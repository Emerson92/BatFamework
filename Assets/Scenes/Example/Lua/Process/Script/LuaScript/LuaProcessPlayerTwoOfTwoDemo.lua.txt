local Self = nil
local PlayerTwo_2 = nil
local StartToExcute = false
local TargetScale
local StartScale
local Timer = 0

function AssetInit(data,target)
    print("lua AssetInit")
	local Player = CS.UnityEngine.Resources.Load(CS.THEDARKKNIGHT.ConstDefine.BFameWorkPathDefine.BFameResourceTestProcessPath .. "/playerTwo_2")
	PlayerTwo_2 = CS.UnityEngine.GameObject.Instantiate(Player)
	Self = target
	Self:ReadyToExcute()
end

function DataInit(data,target)
    print("lua DataInit")
	CS.THEDARKKNIGHT.EventSystem.BEventManager:Instance():AddListener(CS.THEDARKKNIGHT.EventDefine.BatEventDefine.LEFTPRESSEVENT,LeftPressCallbackPlayerTwoTwo)
end

function Destory()
    print("lua Destory")
	CS.UnityEngine.GameObject.Destroy(PlayerTwo_2)
	CS.THEDARKKNIGHT.EventSystem.BEventManager:Instance():RemoveListener(CS.THEDARKKNIGHT.EventDefine.BatEventDefine.LEFTPRESSEVENT,LeftPressCallbackPlayerTwoTwo)
end

function FixedUpdate()
    --print("lua FixedUpdate")
end

function ProcessExcute()
    print("lua ProcessExcute")
end

function StopExcute()
    print("lua StopExcute")
end

function Update()
    if(StartToExcute)
		then
            Timer = (Timer + CS.UnityEngine.Time.deltaTime)
            PlayerTwo_2.transform.localScale = CS.UnityEngine.Vector3.Lerp(StartScale, TargetScale, Timer/5);
        if(PlayerTwo_2.transform.localScale == TargetScale)
		then
            StartToExcute = false;
            Timer = 0;
            Self:ProcessFinish();
		end
	end
end

function LeftPressCallbackPlayerTwoTwo(data)
	print("LeftPressCallback playerTwo_2" )
    local packet = data;
    if (packet.Info.CastGameObject == PlayerTwo_2)
	then
		TargetScale = PlayerTwo_2.transform.localScale * 0.5
		StartScale = PlayerTwo_2.transform.localScale
        StartToExcute = true;
    end
	return nuil;
end
