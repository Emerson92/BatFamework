function AssetInit(data,target)
    print("lua AssetInit")

	Player = CS.UnityEngine.Resources.Load(CS.THEDARKKNIGHT.ConstDefine.BFameWorkPathDefine.BFameResourceTestProcessPath .. "/playerTwo_2")
	PlayerTwo_2 = CS.UnityEngine.GameObject.Instantiate(Player)
	Self = target
	Self:ReadyToExcute()
end

function DataInit(data,target)
    print("lua DataInit")
	CS.THEDARKKNIGHT.EventSystem.BEventManager:Instance():AddListener(CS.THEDARKKNIGHT.EventDefine.BatEventDefine.LEFTPRESSEVENT,LeftPressCallback)
end

function Destory()
    print("lua Destory")
	CS.THEDARKKNIGHT.EventSystem.BEventManager:Instance():RemoveListener(CS.THEDARKKNIGHT.EventDefine.BatEventDefine.LEFTPRESSEVENT,LeftPressCallback)
end

function Update()
    if(StartToExcute)
		then
            Timer = Time + CS.UnityEngine.Time.deltaTime
            PlayerTwo_2.transform.localScale = CS.UnityEngine.Vector3.Lerp(PlayerTwo_2.transform.localScale, TargetScale, Timer/ 5);
        if(PlayerTwo_2.transform.localScale == TargetScale)
		then
            StartToExcute = false;
            Timer = 0;
            ProcessFinish();
		end
	end
end
